        // ==================== RENDERING ====================
        function renderElement(el) {

            const baseStyles = `
                position: absolute;
                left: ${el.x}px;
                top: ${el.y}px;
                width: ${el.width}px;
                height: ${el.height}px;
            `;

            switch(el.type) {
                case 'text':
                    html = `
                        <div class="editable-element text-element" data-id="${el.id}" style="${baseStyles}
                            font-size: ${el.styles.fontSize}px;
                            font-weight: ${el.styles.fontWeight};
                            font-family: '${el.styles.fontFamily}', sans-serif;
                            color: ${el.styles.color};
                            text-align: ${el.styles.textAlign};
                            line-height: ${el.styles.lineHeight};
                            background: ${el.styles.backgroundColor};
                            padding: ${el.styles.padding};

                    `;
                    break;

                case 'image':
                    html = `
                        <div class="editable-element image-element" data-id="${el.id}" style="${baseStyles}
                            border-radius: ${el.styles.borderRadius};

                        </div>

                            background: ${el.styles.backgroundColor};

            document.querySelectorAll('.editable-element').forEach(el => el.remove());

        function updateEmptyStates() {
            const leftElements = elements.filter(e => e.page === 'left').length;
            const rightElements = elements.filter(e => e.page === 'right').length;
            
            document.getElementById('leftEmpty').style.display = leftElements > 0 ? 'none' : 'flex';
            document.getElementById('rightEmpty').style.display = rightElements > 0 ? 'none' : 'flex';

        // ==================== SELECTION & INTERACTION ====================
        function attachElementEvents(el) {
            el.addEventListener('mousedown', handleMouseDown);
            el.addEventListener('touchstart', handleTouchStart, { passive: false });
            
            // Resize handles
            el.querySelectorAll('.resize-handle').forEach(handle => {
                handle.addEventListener('mousedown', handleResizeStart);
                handle.addEventListener('touchstart', handleResizeTouchStart, { passive: false });
            });

            // Double click for text editing
            if (el.classList.contains('text-element')) {
                el.addEventListener('dblclick', () => {
                    el.setAttribute('contenteditable', 'true');
                    el.focus();
                });
                
                el.addEventListener('blur', () => {
                    el.setAttribute('contenteditable', 'false');
                    const elementData = elements.find(e => e.id === el.dataset.id);
                    if (elementData) {
                        elementData.content = el.innerHTML;
                        saveState();
                    }
                });
            }
        }

        function handleMouseDown(e) {
            if (e.target.classList.contains('resize-handle')) return;
            
            e.preventDefault();
            const el = e.currentTarget;
            selectElement(el.dataset.id);
            
            isDragging = true;
            const rect = el.getBoundingClientRect();
            dragOffset.x = e.clientX - rect.left;
            dragOffset.y = e.clientY - rect.top;
        }

        function handleTouchStart(e) {
            if (e.target.classList.contains('resize-handle')) return;
            
            e.preventDefault();
            const el = e.currentTarget;
            selectElement(el.dataset.id);
            
            isDragging = true;
            const rect = el.getBoundingClientRect();
            const touch = e.touches[0];
            dragOffset.x = touch.clientX - rect.left;
            dragOffset.y = touch.clientY - rect.top;
        }

        function handleMouseMove(e) {
            if (!isDragging || !selectedElementId) return;
            
            const elementData = elements.find(el => el.id === selectedElementId);
            if (!elementData) return;

            const page = elementData.page === 'left' ? leftPage : rightPage;
            const pageRect = page.getBoundingClientRect();
            
            let newX = e.clientX - pageRect.left - dragOffset.x;
            let newY = e.clientY - pageRect.top - dragOffset.y;
            
            // Constrain to page bounds
            newX = Math.max(0, Math.min(newX, pageRect.width - elementData.width));
            newY = Math.max(0, Math.min(newY, pageRect.height - elementData.height));
            
            elementData.x = newX;
            elementData.y = newY;
            
            const domEl = page.querySelector(`[data-id="${selectedElementId}"]`);
            if (domEl) {
                domEl.style.left = `${newX}px`;
                domEl.style.top = `${newY}px`;
            }
        }

        function handleTouchMove(e) {
            if (!isDragging || !selectedElementId) return;
            
            const elementData = elements.find(el => el.id === selectedElementId);
            if (!elementData) return;

            const page = elementData.page === 'left' ? leftPage : rightPage;
            const pageRect = page.getBoundingClientRect();
            const touch = e.touches[0];
            
            let newX = touch.clientX - pageRect.left - dragOffset.x;
            let newY = touch.clientY - pageRect.top - dragOffset.y;
            
            newX = Math.max(0, Math.min(newX, pageRect.width - elementData.width));
            newY = Math.max(0, Math.min(newY, pageRect.height - elementData.height));
            
            elementData.x = newX;
            elementData.y = newY;
            
            const domEl = page.querySelector(`[data-id="${selectedElementId}"]`);
            if (domEl) {
                domEl.style.left = `${newX}px`;
                domEl.style.top = `${newY}px`;
            }
        }

        function handleMouseUp() {
            if (isDragging) {
                isDragging = false;
                saveState();
            }
            if (isResizing) {
                isResizing = false;
                currentResizeHandle = null;
                saveState();
            }
        }

        // Resize functionality
        function handleResizeStart(e) {
            e.stopPropagation();
            isResizing = true;
            currentResizeHandle = e.target.className.split(' ')[1];
        }

        function handleResizeTouchStart(e) {
            e.stopPropagation();
            isResizing = true;
            currentResizeHandle = e.target.className.split(' ')[1];
        }

        function handleResizeMove(e) {
            if (!isResizing || !selectedElementId || !currentResizeHandle) return;
            
            const elementData = elements.find(el => el.id === selectedElementId);
            if (!elementData) return;

            const page = elementData.page === 'left' ? leftPage : rightPage;
            const pageRect = page.getBoundingClientRect();
            const domEl = page.querySelector(`[data-id="${selectedElementId}"]`);
            if (!domEl) return;

            const mouseX = e.clientX - pageRect.left;
            const mouseY = e.clientY - pageRect.top;
            const minSize = 30;

            switch(currentResizeHandle) {
                case 'se':
                    elementData.width = Math.max(minSize, mouseX - elementData.x);
                    elementData.height = Math.max(minSize, mouseY - elementData.y);
                    break;
                case 'sw':
                    const newWidthSW = Math.max(minSize, elementData.x + elementData.width - mouseX);
                    elementData.x = elementData.x + elementData.width - newWidthSW;
                    elementData.width = newWidthSW;
                    elementData.height = Math.max(minSize, mouseY - elementData.y);
                    break;
                case 'ne':
                    elementData.width = Math.max(minSize, mouseX - elementData.x);
                    const newHeightNE = Math.max(minSize, elementData.y + elementData.height - mouseY);
                    elementData.y = elementData.y + elementData.height - newHeightNE;
                    elementData.height = newHeightNE;
                    break;
                case 'nw':
                    const newWidthNW = Math.max(minSize, elementData.x + elementData.width - mouseX);
                    const newHeightNW = Math.max(minSize, elementData.y + elementData.height - mouseY);
                    elementData.x = elementData.x + elementData.width - newWidthNW;
                    elementData.y = elementData.y + elementData.height - newHeightNW;
                    elementData.width = newWidthNW;
                    elementData.height = newHeightNW;
                    break;
            }

            domEl.style.width = `${elementData.width}px`;
            domEl.style.height = `${elementData.height}px`;
            domEl.style.left = `${elementData.x}px`;
            domEl.style.top = `${elementData.y}px`;
        }

        // Global event listeners for dragging/resizing
        document.addEventListener('mousemove', (e) => {
            handleMouseMove(e);
            handleResizeMove(e);
        });
        document.addEventListener('touchmove', (e) => {
            handleTouchMove(e);
            handleResizeMove(e.touches[0]);
        }, { passive: false });
        document.addEventListener('mouseup', handleMouseUp);

        function updateProperty(prop, value) {
            const el = elements.find(e => e.id === selectedElementId);
            if (!el) return;
            
            el[prop] = parseFloat(value);
            
            const domEl = document.querySelector(`[data-id="${selectedElementId}"]`);
            if (domEl) {
                domEl.style[prop] = value + 'px';
            }
            
            saveState();
        }

        function updateStyle(prop, value) {
            const el = elements.find(e => e.id === selectedElementId);
            if (!el) return;
            
            el.styles[prop] = value;
            
            const domEl = document.querySelector(`[data-id="${selectedElementId}"]`);
            if (domEl) {
                domEl.style[prop] = value;
            }
            
            updatePropertiesPanel();
            saveState();
        }

        function updateHighlightTitle(value) {
            const el = elements.find(e => e.id === selectedElementId);
            if (!el) return;
            el.title = value;
            
            const domEl = document.querySelector(`[data-id="${selectedElementId}"]`);
            if (domEl) {
                const titleEl = domEl.querySelector('div:first-child');
                if (titleEl) titleEl.textContent = value;
            }
            saveState();
        }

        function updateHighlightContent(value) {
            const el = elements.find(e => e.id === selectedElementId);
            if (!el) return;
            el.content = value;
            
            const domEl = document.querySelector(`[data-id="${selectedElementId}"]`);
            if (domEl) {
                const contentEl = domEl.querySelector('div:last-child');
                if (contentEl) contentEl.textContent = value;
            }
            saveState();
        }

        function init() {
            // Load saved project if exists
            const saved = localStorage.getItem('magazineProject');
            if (saved) {
                try {
                    const project = JSON.parse(saved);
                    if (project.elements && project.elements.length > 0) {
                        elements = project.elements;
                        renderAllElements();
                        saveState(); // Initialize history
                        showToast('تم تحميل المشروع المحفوظ');
                    }
                } catch (e) {
                    console.log('No valid saved project');
                }
            }
            
            // Initial state save
            if (elements.length === 0) {
                saveState();
            }
            
            updateEmptyStates();
        }

        init();
    </script>
